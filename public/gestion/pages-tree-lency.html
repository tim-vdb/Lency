📋 Structure simplifiée (Pages uniquement)

src/app/(pages)/
│
├── (public)/
│   ├── (app)/
│   │   ├── page.tsx           # Landing / Accueil
│   │   ├── dashboard/
│   │   │   └── page.tsx         # Marketplace / Projets publics
│   │   └── project/[id]/
│   │       └── page.tsx         # Project Detail
│   │
│   └── auth/
│       ├── signin/page.tsx    # Sign In
│       ├── signup/page.tsx    # Sign Up
│       └── verify/page.tsx    # Email OTP Verify
│
├── account/              # Pages privées (auth required)
│   ├── page.tsx            # Dashboard
│   ├── profile/
│   │   └── page.tsx         # Edit Profile
│   ├── settings/
│   │   └── page.tsx         # User Settings
│   ├── projects/
│   │   ├── page.tsx         # My Projects
│   │   └── [id]/
│   │       ├── page.tsx       # Edit Project
│   │       └── applications/
│   │           └── page.tsx   # View Applications
│   └── applications/
│       └── page.tsx         # My Applications
│
└── admin/                 # Pages admin (admin role required)
    ├── page.tsx            # Admin Dashboard
    ├── users/
    │   ├── page.tsx         # User Management
    │   └── [id]/
    │       └── page.tsx       # User Detail
    ├── projects/
    │   └── page.tsx         # Project Moderation
    ├── reports/
    │   └── page.tsx         # Report Review
    └── analytics/
        └── page.tsx         # Site Analytics

🔑 Clés de navigation

// 🌐 Pages PUBLIQUES (accessible à tous)
/                          # Landing
/dashboard                 # Marketplace / Voir projets
/project/[id]              # Détail d'un projet
/auth/signin               # Connexion
/auth/signup               # Inscription

// 👤 Pages PRIVÉES (auth required)
/account                   # Dashboard utilisateur
/account/profile           # Éditer profil
/account/settings          # Paramètres compte
/account/projects          # Mes projets
/account/projects/[id]     # Éditer un projet
/account/projects/[id]/applications  # Candidatures pour mon projet
/account/applications      # Mes candidatures

// 🔐 Pages ADMIN (admin role required)
/admin                     # Admin Dashboard
/admin/users               # Gestion utilisateurs
/admin/users/[id]          # Détail utilisateur
/admin/projects            # Modération projets
/admin/reports             # Examiner rapports
/admin/analytics           # Statistiques site
